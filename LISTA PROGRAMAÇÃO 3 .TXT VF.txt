  j=0;
  while(j<strlen(str)){
    if(str[j]>=48 && str[j]<=57){
      mostrar[t] = str[j];
        t++;
    }
    j++;
  }
  g = strlen(mostrar);
  int vetor[g];
  for(j=0; j<g; j++){
    vetor[j] = mostrar[j]-48;
  }
  for(j=0; j<g; j++){
    if(vetor[j] > maiorNum){
      maiorNum = vetor[j];
    }
  }
  x = maiorNum;
  while( x > 0){
    for (y=0; y < g; y++){
      if (vetor[y] >= x){
        printf("*");
      }
      else{
        printf(" ");
      }
    }
    x--;
    printf("\n");
    }
}
